from altair import *

Chart('http://vega.github.io/vega-lite/data/cars.json').mark_text(
    applyColorToBackground=True,
).encode(
    color='mean(Horsepower):Q',
    column='Cylinders:O',
    row='Origin:O',
    text='count(*):Q',
)
