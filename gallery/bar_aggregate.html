Aggregate Bar Chart¶

< Simple Bar Chart | Example Gallery | Grouped Bar Chart >

from altair import *

Chart('http://vega.github.io/vega-lite/data/population.json',
    description='A bar chart showing the US population distribution of age groups in 2000.',
).mark_bar().encode(
    x=X('sum(people):Q',
        axis=Axis(
            title='population',
        ),
    ),
    y=Y('age:O',
        scale=Scale(
            bandSize=17.0,
        ),
    ),
).transform_data(
    filter='datum.year == 2000',
)
