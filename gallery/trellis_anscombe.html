Anscombe’s Quartet¶

< Strip Plot | Example Gallery | Trellis Bar Chart >

from altair import *

Chart('http://vega.github.io/vega-lite/data/anscombe.json',
    description="Anscombe's Quartet",
).mark_circle(
    opacity=1.0,
).encode(
    column='Series:N',
    x=X('X:Q',
        scale=Scale(
            zero=False,
        ),
    ),
    y=Y('Y:Q',
        scale=Scale(
            zero=False,
        ),
    ),
)
