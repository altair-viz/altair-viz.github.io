from altair import *

Chart('http://vega.github.io/vega-lite/data/cars.json').mark_bar().encode(
    row='Origin:N',
    x=X('Horsepower:Q',
        bin=Bin(
            maxbins=15.0,
        ),
    ),
    y='count(*):Q',
)
