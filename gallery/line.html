Line Chart¶

< Histogram | Example Gallery | Multi Series Line Chart >

from altair import *

Chart(Data(
    format=DataFormat(
        type='csv',
    ),
    url='http://vega.github.io/vega-lite/data/stocks.csv',
),
    description="Google's stock price over time.",
).mark_line().encode(
    x='date:T',
    y='price:Q',
).transform_data(
    filter="datum.symbol==='GOOG'",
)
